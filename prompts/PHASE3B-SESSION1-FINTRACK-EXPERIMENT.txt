```go
// In fintrack's internal/context/tracker.go (hypothetical)
package context

import (
    mycontext "github.com/jefferycaldwell/my-context-copilot/pkg/client"
)

// Tracker wraps my-context operations
type Tracker struct {
    client *mycontext.Client
    sessionID string
}

// NewTracker creates a new context tracker
func NewTracker() (*Tracker, error) {
    client, err := mycontext.NewClient()
    if err != nil {
        return nil, err
    }

    sessionID := fmt.Sprintf("fintrack-%s", time.Now().Format("2006-01-02"))
    if err := client.Start(sessionID); err != nil {
        return nil, err
    }

    return &Tracker{client: client, sessionID: sessionID}, nil
}

// TrackOperation logs an operation to my-context
func (t *Tracker) TrackOperation(operation string, details map[string]interface{}) error {
    note := fmt.Sprintf("Operation: %s - %v", operation, details)
    return t.client.Note(note)
}
```
